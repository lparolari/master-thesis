% Python listing style
%
% Resources:
% * https://tex.stackexchange.com/a/83883
% * https://nasa.github.io/nasa-latex-docs/html/examples/listing.html

\definecolor{codebackgroundcolor}{rgb}{0.95,0.95,0.92}
\definecolor{codestringcolor}{rgb}{0,0.5,0}

\lstdefinestyle{simplepython}{
  language=Python,
  basicstyle=\ttfamily\footnotesize,
  backgroundcolor=\color{codebackgroundcolor},
  stringstyle=\color{codestringcolor},
  breaklines=true,
  postbreak=\mbox{\textcolor{red}{$\hookrightarrow$}\space},
  showspaces=false,
  showstringspaces=false,
  showtabs=false,
  numbers=left,
  numbersep=5pt
}
